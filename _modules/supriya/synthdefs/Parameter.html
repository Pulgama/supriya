    ### INITIALIZER ###

    def __init__(
        self,
        lag=None,
        name=None,
        parameter_rate=ParameterRate.CONTROL,
        range_=None,
        unit=None,
        value=None,
    ):
        import supriya.synthdefs

        # assert name
        if lag is not None:
            lag = float(lag)
        self._lag = lag
        if name is not None:
            name = str(name)
        self._name = name
        self._parameter_rate = ParameterRate.from_expr(parameter_rate)
        if range_ is not None:
            assert isinstance(range_, supriya.synthdefs.Range)
        self._range = range_
        self._unit = unit
        self._uuid = None
        if isinstance(value, collections.Sequence):
            value = tuple(float(_) for _ in value)
            assert value, value
        else:
            value = float(value)
        self._value = value
